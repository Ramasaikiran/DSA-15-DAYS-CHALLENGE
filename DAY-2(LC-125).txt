class Solution {
    public boolean isPalindrome(String s) {
      String str = s.toLowerCase().replaceAll("[^a-z0-9]","");
        if(str.length()<=1){
            return true;
        }
        int l=0,r=str.length()-1;
            while(l<r){
                if(str.charAt(l)!=str.charAt(r)){
                    return false;
                }
                l++;
                r--;
        }
        return true;
    }
}
