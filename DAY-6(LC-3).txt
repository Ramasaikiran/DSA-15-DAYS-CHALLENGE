class Solution {
    public int lengthOfLongestSubstring(String s) {
        int l=0,max=0;
        Set<Character>hm=new HashSet<>();
        for(int r=0;r<s.length();r++){
        while(hm.contains(s.charAt(r))){
            hm.remove(s.charAt(l));
            l++;
        }
        hm.add(s.charAt(r));
        max=Math.max(max,r-l+1);
    }
    return max;
    }
}
